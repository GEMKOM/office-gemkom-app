Interactive Example (Clickable Cards)

Click on the cards to see the interaction!

// Interactive cards with click handlers const interactiveStats = new StatisticsCards('interactive-stats', { cards: [ { title: 'Click to View Details', value: '42', icon: 'fas fa-hand-pointer', color: 'primary', onClick: 'handleCardClick(0)', tooltip: 'Click to view user details' }, { title: 'Click to Export', value: '156', icon: 'fas fa-download', color: 'success', onClick: 'handleCardClick(1)', tooltip: 'Click to export data' }, { title: 'Click to Refresh', value: '89', icon: 'fas fa-sync-alt', color: 'info', onClick: 'handleCardClick(2)', tooltip: 'Click to refresh data' } ] }); function handleCardClick(index) { alert(`Card ${index} clicked! You can implement custom logic here.`); }

Dynamic Updates Example

// Dynamic updates example const dynamicStats = new StatisticsCards('dynamic-stats', { cards: [ { title: 'Random Value 1', value: '0', icon: 'fas fa-dice-one', color: 'primary' }, { title: 'Random Value 2', value: '0', icon: 'fas fa-dice-two', color: 'success' }, { title: 'Random Value 3', value: '0', icon: 'fas fa-dice-three', color: 'warning' } ] }); function updateRandomValues() { dynamicStats.updateValues({ 0: Math.floor(Math.random() * 100), 1: Math.floor(Math.random() * 100), 2: Math.floor(Math.random() * 100) }); } function addNewCard() { const cardCount = dynamicStats.options.cards.length; dynamicStats.addCard({ title: `New Card ${cardCount + 1}`, value: Math.floor(Math.random() * 100), icon: 'fas fa-plus', color: 'info' }); } function removeLastCard() { const cardCount = dynamicStats.options.cards.length; if (cardCount > 0) { dynamicStats.removeCard(cardCount - 1); } }

Loading and Empty States

// Loading and empty states const stateStats = new StatisticsCards('state-stats', { cards: [ { title: 'Sample Data', value: '100', icon: 'fas fa-chart-bar', color: 'primary' } ] }); function showLoading() { stateStats.showLoading(); } function showEmpty() { stateStats.showEmpty('No data available at the moment'); } function showData() { stateStats.setCards([ { title: 'Sample Data', value: '100', icon: 'fas fa-chart-bar', color: 'primary' } ]); }
